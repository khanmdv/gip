"use strict";
 
// Lib Imports
var React = require("react");
var marked = require("marked");
 
/**
 * Renders a HTML Node from a string. Uses GitHub markdown to convert markdown
 * to HTML.
 * 
 */
var HtmlNode = React.createClass({
    displayName: "HtmlNode",
 
    /** 
     * Properties
     * 
     * {html} HTML Code as string.
     */
    PropTypes: {
        html: React.PropTypes.string
    },
 
    /**
     * Renders marked down content.
     */
    render: function render() {
        var self = this;
        var html = this.props.html;
 
        return React.createElement("div", { className: "html-node", dangerouslySetInnerHTML: {
                __html: marked(new String(html), {
                    renderer: new marked.Renderer(),
                    gfm: true,
                    tables: true,
                    breaks: true,
                    pedantic: false,
                    sanitize: true,
                    smartLists: true,
                    smartypants: false
                })
            } });
    }
});
 
module.exports = HtmlNode;
