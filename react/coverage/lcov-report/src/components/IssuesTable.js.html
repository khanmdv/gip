"use strict";
 
// Lib Imports
var React = require("react");
 
// Our Imports
var IssueCell = require("./IssueCell");
 
/**
 * Renders the Issues table.
 */
var IssuesTable = React.createClass({
    displayName: "IssuesTable",
 
    /**
     * Properties
     * 
     * {issues} List of issues to render
     */
    PropTypes: {
        issues: React.PropTypes.array
    },
 
    /**
     * Render Issue rows
     */
    _renderIssueRows: function _renderIssueRows() {
        return this.props.issues.map(function (issue, i) {
            return React.createElement(
                "tr",
                { key: issue.number, className: "issue-table-rows" },
                React.createElement(
                    "td",
                    null,
                    React.createElement(IssueCell, { key: i, issue: issue })
                )
            );
        });
    },
    /**
     * Render issues.
     */
    render: function render() {
        var self = this;
 
        return React.createElement(
            "table",
            { className: "table table-hover issues-main-table" },
            React.createElement(
                "thead",
                null,
                React.createElement(
                    "tr",
                    null,
                    React.createElement(
                        "td",
                        null,
                        React.createElement(
                            "b",
                            null,
                            "Issues"
                        )
                    )
                )
            ),
            React.createElement(
                "tbody",
                null,
                self._renderIssueRows()
            )
        );
    }
});
 
module.exports = IssuesTable;
