"use strict";
 
// Lib Imports
var React = require("react");
var Router = require("react-router");
var Link = Router.Link;
 
// Our Imports
var GitIssue = require("../models/GitIssue");
var HtmlNode = require("./HtmlNode");
 
/**
 * Represents a table cell. Renders issue details.
 */
var IssueCell = React.createClass({
    displayName: "IssueCell",
 
    /**
     * Properties
     * 
     * {issue} GitIssue object
     */
    PropTypes: {
        issue: React.PropTypes.object
    },
    _renderLabels: function _renderLabels() {
        return this.props.issue.labels.map(function (label, i) {
            var style = {
                backgroundColor: "#" + label.color,
                marginRight: "3px"
            };
            return React.createElement(
                "span",
                { className: "issue-label", style: style, key: i },
                React.createElement(
                    "a",
                    { className: "label", href: label.url },
                    label.name
                )
            );
        });
    },
    render: function render() {
        var issueStateIconColor = this.props.issue.state === GitIssue.IssueState.Open ? "orange" : "green";
        var self = this;
        return React.createElement(
            "div",
            { className: "issue-cell" },
            React.createElement(
                "h4",
                { className: "issue-title" },
                React.createElement(
                    "span",
                    { className: "text-muted" },
                    "#",
                    this.props.issue.number
                ),
                "  ",
                React.createElement(
                    Link,
                    { to: "/issue/" + this.props.issue.number },
                    this.props.issue.title
                )
            ),
            " ",
            self._renderLabels(),
            React.createElement(
                "div",
                { className: "issue-cell-user" },
                React.createElement("img", { className: "issue-gravatar", src: this.props.issue.user.gravatarURL }),
                " ",
                React.createElement(
                    "a",
                    { className: "user-link text-muted", href: this.props.issue.user.profileURL },
                    this.props.issue.user.userName
                ),
                " ",
                React.createElement(
                    "span",
                    { className: "text-muted" },
                    "created ",
                    this.props.issue.createdAt.fromNow()
                )
            ),
            React.createElement("br", null),
            React.createElement(
                "div",
                { className: "issue-summary" },
                React.createElement(
                    "div",
                    { style: { display: "inline-block" } },
                    React.createElement(HtmlNode, { html: this.props.issue.getShortSummary() })
                ),
                " ",
                React.createElement(
                    "i",
                    { style: { display: "inline-block" }, className: "fa fa-ellipsis-h" },
                    " "
                )
            )
        );
    }
});
 
module.exports = IssueCell;
