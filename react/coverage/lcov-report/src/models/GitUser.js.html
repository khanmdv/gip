"use strict";
 
// Our Imports
var utils = require("../utils");
 
/**
 * GitUser Class - Represents a user.
 */
var GitUser = function GitUser() {};
 
/**
 * Class Methods
 */
utils.extend(GitUser, {
 
    /**
     * Mapping between REST API and class properties.
     */
    JSONKeyValueMap: function JSONKeyValueMap() {
        return {
            userName: "login",
            gravatarURL: "avatar_url",
            profileURL: "html_url"
        };
    }
});
 
/**
 * Instance Methods
 */
GitUser.prototype = {
    toString: function toString() {
        return JSON.stringify({
            gravatarURL: this.gravatarURL,
            userName: this.userName,
            profileURL: this.profileURL
        }, null, 4);
    }
};
 
module.exports = GitUser;
