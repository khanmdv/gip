var utils = require("../utils");
 
/**
 * GitLabel Class - Represents a label.
 */
var GitLabel = function GitLabel() {};
 
/**
 * Class Methods
 */
utils.extend(GitLabel, {
 
    /**
     * Mapping between REST API and class properties.
     */
    JSONKeyValueMap: function JSONKeyValueMap() {
        return {
            url: "url",
            name: "name",
            color: "color"
        };
    }
});
 
/**
 * Instance Methods
 */
GitLabel.prototype = {
    toString: function toString() {
        return JSON.stringify({
            url: this.url,
            name: this.name,
            color: this.color
        }, null, 4);
    }
};
